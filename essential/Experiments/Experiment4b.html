                                          $scope.getweather = function () {
                                            var address = URL.replace("CITY", $scope.cityname);
                                            $http.get(address).success(function (response) {
                                              $scope.Weather = response.data.weather;
                                            console.log($scope.Weather);
                                                     });}}
